\begin{algorithm}
\caption{Earliest Job First}
\begin{algorithmic}
\PROCEDURE{EarliestJobFirst}{$I$}
\STATE Accept the earliest starting job $j$ from $I$ that does not overlap any previously accepted job.
\STATE Repeat until no more such jobs remain.
\ENDPROCEDURE
\end{algorithmic}
\end{algorithm}
