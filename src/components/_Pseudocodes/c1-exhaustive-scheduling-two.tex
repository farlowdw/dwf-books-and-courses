\begin{algorithm}
\caption{Exhaustive Scheduling (Revised)}
\begin{algorithmic}
\PROCEDURE{ExhaustiveScheduling}{$I$}
\STATE Test all $2^n$ subsets of intervals from $I$.
\RETURN the largest subset consisting of mutually non-overlapping intervals.
\ENDPROCEDURE
\end{algorithmic}
\end{algorithm}
